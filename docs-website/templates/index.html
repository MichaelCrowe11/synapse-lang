📝 Example: Quantum Bell State

from synapse_lang.quantum_designer import QuantumCircuit, QuantumSimulator

# Create a 2-qubit circuit
circuit = QuantumCircuit(2)

# Create Bell state (entanglement)
circuit.add_gate("H", [0])      # Hadamard on qubit 0
circuit.add_gate("CNOT", [0, 1])  # CNOT with 0 as control

# Simulate the circuit
simulator = QuantumSimulator()
result = simulator.simulate(circuit)

print("Quantum state:", result)
# Output: {'00': 0.707+0j, '11': 0.707+0j}

🔬 Scientific Type Inference

from synapse_lang.type_inference import TypeInference

# Automatic type inference with uncertainty
type_system = TypeInference()

# Infer types for scientific computations
matrix_type = type_system.infer_matrix_type([[1, 2], [3, 4]])
print(f"Matrix type: {matrix_type}")  # Matrix[Int, 2, 2]

# Handle uncertain measurements
measurement = 3.14 ± 0.01  # Uncertain type
uncertain_type = type_system.infer_uncertain_type(measurement)
print(f"Measurement type: {uncertain_type}")  # Uncertain[Float, Float]

from synapse_lang.collaboration import CollaborationManager
