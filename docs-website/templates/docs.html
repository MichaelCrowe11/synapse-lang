Get started with Synapse Language in under 5 minutes!

1. Installation

# Using pip (Python 3.8+)
pip install synapse_lang

# Using npm

2. Your First Quantum Circuit

from synapse_lang.quantum_designer import QuantumCircuit, QuantumSimulator

# Create a 2-qubit circuit
circuit = QuantumCircuit(2)

# Add quantum gates
circuit.add_gate("H", [0])      # Hadamard gate
circuit.add_gate("CNOT", [0, 1])  # Controlled-NOT

# Simulate
simulator = QuantumSimulator()
result = simulator.simulate(circuit)
print("Quantum state:", result)

Quantum Teleportation

from synapse_lang.quantum_designer import QuantumCircuit, QuantumSimulator

def quantum_teleportation():
    # Create 3-qubit circuit for teleportation
    circuit = QuantumCircuit(3)

    # Prepare entangled pair (Bell state)
    circuit.add_gate("H", [1])
    circuit.add_gate("CNOT", [1, 2])

    # Prepare state to teleport on qubit 0
    circuit.add_gate("H", [0])

    # Bell measurement
    circuit.add_gate("CNOT", [0, 1])
    circuit.add_gate("H", [0])

    return circuit

Distributed Word Count

from synapse_lang.distributed import DistributedComputing
import asyncio

async def distributed_word_count(documents):
    dc = DistributedComputing()

    def map_words(doc):
        return [(word.lower(), 1) for word in doc.split()]

    def reduce_counts(word, counts):
        return (word, sum(counts))

    return await dc.map_reduce(documents, map_words, reduce_counts)
